## 전체 구조 개요

관리자가 컨텐츠 작성 시 `answer_type`으로 템플릿 종류를 선택하면,  
해당 템플릿에 필요한 추가 설정값이 `template_config` 컬럼(JSON 문자열)에 저장된다.  
사용자 화면은 `answer_type` 값에 따라 해당 템플릿 UI를 렌더링한다.

```
answer_type 값       템플릿
────────────────     ──────────────────────────────
"yes_no"             기존 Yes/No 버튼
"text"               기존 자유 텍스트 입력
"choice"             템플릿 1 — 단일/다중 선택형
"car_number"         템플릿 2 — 차량번호 입력
"plan_select"        템플릿 3 — 실손 보상 안 선택
"education"          템플릿 4 — 교육자료 배포 및 열람 확인
```

```python
from pydantic import BaseModel

class ContentCreate(BaseModel):
    title:         str
    target_emp_id: str
    body_1:        str
    body_2:        str | None = None
    body_3:        str | None = None
    answer_type:   Literal[

class Content(ContentCreate):

    updated_at: str | None = None
```

### `app/repositories/excel_repository.py` — `create_content` 저장 시 JSON 직렬화

```python
import json

def create_content(self, payload, creator_id: str) -> dict:
    with _excel_lock:
        df = self._load("contents")
        new_row = {
            "content_id":      str(uuid.uuid4()),
            "title":           payload.title,
            "target_emp_id":   payload.target_emp_id,
            "body_1":          payload.body_1,
            "body_2":          payload.body_2 or "",
            "body_3":          payload.body_3 or "",
            "answer_type":     payload.answer_type,
            "template_config": json.dumps(payload.template_config, ensure_ascii=False),
            "file_path":       "",          # 파일 업로드는 별도 엔드포인트 처리
            "created_by":      creator_id,
            "created_at":      datetime.now().isoformat(),
            "updated_at":      "",
        }
        df = pd.concat([df, pd.DataFrame([new_row])], ignore_index=True)
        self._save("contents", df)
        return new_row

def get_contents_by_emp_id(self, emp_id: str) -> list[dict]:
    df = self._load("contents")
    df_conf = self._load("confirmations")

    rows = df[df["target_emp_id"] == emp_id].to_dict(orient="records")
    for row in rows:
        # template_config JSON 파싱
        try:
            row["template_config"] = json.loads(row.get("template_config") or "{}")
        except (json.JSONDecodeError, TypeError):
            row["template_config"] = {}

        # 확인 여부 병합
        conf = df_conf[
            (df_conf["content_id"] == row["content_id"]) &
            (df_conf["emp_id"] == emp_id)
        ]
        row["is_confirmed"]   = not conf.empty and str(conf.iloc[0]["is_confirmed"]).upper() == "TRUE"
        row["confirmed_text"] = conf.iloc[0]["confirmed_text"] if not conf.empty else ""
    return rows
```

### `app/routers/admin.py` — 파일 업로드 추가

```python
import shutil, os
from fastapi import UploadFile, File, Form
from pathlib import Path

UPLOAD_DIR = Path("data/uploads")
UPLOAD_DIR.mkdir(parents=True, exist_ok=True)

@router.post("/api/admin/contents/{content_id}/upload")
async def upload_education_file(
    content_id: str,
    file: UploadFile = File(...),
    _=Depends(require_admin_user)
):
    # 허용 확장자 검사
    allowed = {".pdf", ".png", ".jpg", ".jpeg", ".pptx", ".docx", ".xlsx"}
    ext = Path(file.filename).suffix.lower()
    if ext not in allowed:
        raise HTTPException(status_code=400, detail=f"허용되지 않는 파일 형식: {ext}")

    # 파일명: content_id + 원본 확장자 (경로 추측 방지)
    save_name = f"{content_id}{ext}"
    save_path = UPLOAD_DIR / save_name

    with save_path.open("wb") as f:
        shutil.copyfileobj(file.file, f)

    # Excel에 file_path 업데이트
    content_repo.update_file_path(content_id, str(save_path))

    return {"status": "ok", "filename": save_name}
```

### 파일 다운로드 (인증 세션 전용)

```python
# app/routers/content.py
from fastapi.responses import FileResponse

@router.get("/api/content/{content_id}/file")
async def download_education_file(
    content_id: str,
    current_user=Depends(require_authenticated_user)
):
    content = content_repo.get_content_by_id(content_id)
    if not content or content["answer_type"] != "education":
        raise HTTPException(status_code=404)
    if content["target_emp_id"] != current_user["sub"]:
        raise HTTPException(status_code=403)

    path = Path(content["file_path"])
    if not path.exists():
        raise HTTPException(status_code=404, detail="파일을 찾을 수 없습니다.")

    # 열람 이벤트 즉시 기록 (클릭 = 확인 처리)
    content_repo.set_confirmation(
        content_id=content_id,
        emp_id=current_user["sub"],
        confirmed=True,
        confirmed_text="열람 완료"
    )

    return FileResponse(
        path=str(path),
        filename=path.name,
        media_type="application/octet-stream"
    )
```

### `excel_repository.py` — 추가 메서드 2개

```python
def get_content_by_id(self, content_id: str) -> dict | None:
    df = self._load("contents")
    row = df[df["content_id"] == content_id]
    if row.empty:
        return None
    r = row.iloc[0].to_dict()
    try:
        r["template_config"] = json.loads(r.get("template_config") or "{}")
    except (json.JSONDecodeError, TypeError):
        r["template_config"] = {}
    return r

def update_file_path(self, content_id: str, file_path: str):
    with _excel_lock:
        df = self._load("contents")
        df.loc[df["content_id"] == content_id, "file_path"] = file_path
        self._save("contents", df)
```

## PATCH 4 — `admin.js`: 템플릿 설정 로직

```javascript
// ── 템플릿 카드 선택 강조 ──
function onAnswerTypeChange() {
  const val = document.querySelector('input[name="answer_type"]:checked')?.value;

  // 카드 테두리 강조
  document.querySelectorAll(".tpl-card").forEach(card => {
    card.style.borderColor = card.dataset.val === val ? "#3b5bdb" : "#dde1ee";
    card.style.background  = card.dataset.val === val ? "#f0f4ff" : "#fff";
  });

  // 추가 설정 패널 표시/숨김
  document.querySelectorAll(".tpl-config").forEach(el => el.classList.add("hidden"));
  const panelMap = { choice: "config-choice", plan_select: "config-plan", education: "config-education" };
  if (panelMap[val]) document.getElementById(panelMap[val])?.classList.remove("hidden");
}

// ── [T1] 선택형 옵션 관리 ──
let _choiceCount = 0;

function addChoiceOption(value = "") {
  _choiceCount++;
  const id = `choice-opt-${_choiceCount}`;
  const wrap = document.createElement("div");
  wrap.id = id;
  wrap.style = "display:flex; gap:8px; align-items:center;";
  wrap.innerHTML = `
    <input type="text" placeholder="선택지 ${_choiceCount}" value="${value}"
           style="flex:1; padding:8px 10px; border:1px solid #dde1ee;
                  border-radius:6px; font-size:13px;"
           data-choice-input>
    <button type="button" onclick="document.getElementById('${id}').remove()"
            style="padding:6px 10px; border:1px solid #fca5a5; border-radius:6px;
                   background:#fff; color:#ef4444; cursor:pointer;">✕</button>
  `;
  document.getElementById("choice-option-list").appendChild(wrap);
}

function getChoiceConfig() {
  const mode = document.querySelector('input[name="choice_mode"]:checked')?.value ?? "single";
  const options = [...document.querySelectorAll("[data-choice-input]")]
    .map(el => el.value.trim()).filter(Boolean);
  return { mode, options };
}

// ── [T3] 실손 안 관리 ──
let _planCount = 0;

function addPlanOption(title = "", desc = "") {
  _planCount++;
  const id = `plan-opt-${_planCount}`;
  const n  = _planCount;
  const wrap = document.createElement("div");
  wrap.id = id;
  wrap.style = "border:1px solid #dde1ee; border-radius:8px; padding:12px 16px;";
  wrap.innerHTML = `
    <div style="display:flex; justify-content:space-between; margin-bottom:8px;">
      <span style="font-weight:600; font-size:13px; color:#3b5bdb;">${n}안</span>
      <button type="button" onclick="document.getElementById('${id}').remove()"
              style="border:none; background:none; color:#ef4444; cursor:pointer;">✕</button>
    </div>
    <input type="text" placeholder="${n}안 제목" value="${title}"
           style="width:100%; padding:8px 10px; border:1px solid #eee;
                  border-radius:6px; font-size:13px; margin-bottom:6px;"
           data-plan-title>
    <textarea placeholder="${n}안 세부 설명 (선택)" rows="2"
              style="width:100%; padding:8px 10px; border:1px solid #eee;
                     border-radius:6px; font-size:13px; resize:vertical;"
              data-plan-desc>${desc}</textarea>
  `;
  document.getElementById("plan-option-list").appendChild(wrap);
}

function getPlanConfig() {
  const plans = [];
  document.querySelectorAll("#plan-option-list > div").forEach((el, i) => {
    plans.push({
      label: `${i + 1}안`,
      title: el.querySelector("[data-plan-title]")?.value.trim() ?? "",
      desc:  el.querySelector("[data-plan-desc]")?.value.trim() ?? "",
    });
  });
  return { plans };
}

// ── 제출 시 template_config 조합 ──
async function submitCreateContent() {
  const title      = document.getElementById("content-title").value.trim();
  const body1      = document.getElementById("content-body1").value.trim();
  const answerType = document.querySelector('input[name="answer_type"]:checked')?.value;
  const targets    = [...document.querySelectorAll(".user-checkbox:checked")].map(cb => cb.value);

  if (!title)  { alert("제목을 입력하세요."); return; }
  if (!body1)  { alert("내용을 입력하세요."); return; }
  if (targets.length === 0) { alert("대상자를 1명 이상 선택하세요."); return; }

  // 템플릿별 config 구성
  let templateConfig = {};
  if (answerType === "choice")     templateConfig = getChoiceConfig();
  if (answerType === "plan_select") templateConfig = getPlanConfig();
  if (answerType === "education") {
    const link = document.getElementById("edu-link-input").value.trim();
    if (link) templateConfig = { link };
    // 파일은 컨텐츠 생성 후 별도 업로드
  }

  // 유효성: 선택형 옵션 최소 2개
  if (answerType === "choice" && templateConfig.options?.length < 2) {
    alert("선택지를 2개 이상 입력하세요."); return;
  }
  // 유효성: 실손 안 최소 2개
  if (answerType === "plan_select" && templateConfig.plans?.length < 2) {
    alert("보상 안을 2개 이상 입력하세요."); return;
  }

  const created = await Promise.all(targets.map(async emp_id => {
    const res = await fetch("/api/admin/contents", {
      method: "POST",
      credentials: "include",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        title, body_1: body1,
        target_emp_id: emp_id,
        answer_type: answerType,
        template_config: templateConfig,
      }),
    });
    const data = await res.json();
    return { ok: res.ok, content_id: data.content_id, emp_id };
  }));

  // 템플릿 4: 파일 업로드 (생성된 content_id별)
  const eduFile = document.getElementById("edu-file-input")?.files[0];
  if (answerType === "education" && eduFile) {
    await Promise.all(created.filter(c => c.ok).map(async ({ content_id }) => {
      const fd = new FormData();
      fd.append("file", eduFile);
      await fetch(`/api/admin/contents/${content_id}/upload`, {
        method: "POST", credentials: "include", body: fd,
      });
    }));
  }

  const failed = created.filter(c => !c.ok).length;
  alert(failed > 0
    ? `${targets.length}명 중 ${failed}명 생성 실패.`
    : `${targets.length}명에게 컨텐츠가 생성되었습니다.`);

  closeCreateModal();
  await refreshContentTable();
}
```

## PATCH 6 — `content.js`: 4종 핸들러

```javascript
// ── [T1] 선택형 제출 ──
async function handleChoiceSubmit(contentId) {
  const wrap = document.getElementById(`choice-${contentId}`);
  const mode = wrap.dataset.mode;
  let selected = [];

  if (mode === "single") {
    const checked = wrap.querySelector(`input[name="choice-${contentId}"]:checked`);
    if (!checked) { alert("항목을 선택하세요."); return; }
    selected = [checked.value];
  } else {
    selected = [...wrap.querySelectorAll(`input[name="choice-${contentId}"]:checked`)]
                .map(el => el.value);
    if (selected.length === 0) { alert("하나 이상 선택하세요."); return; }
  }

  const text = selected.join(", ");
  await _submitConfirm(contentId, "choice", text);
}

// ── [T2] 차량번호 ──
function carInputFilter(el, type) {
  if (type === "digit")     el.value = el.value.replace(/[^0-9]/g, "");
  if (type === "hangul")    el.value = el.value.replace(/[^\uAC00-\uD7A3]/g, "");
  if (type === "alpha-num") el.value = el.value.replace(/[^0-9A-Za-z\uAC00-\uD7A3가-힣]/g, "");
}

function carAutoFocus(el, nextId) {
  if (el.value.length >= el.maxLength) {
    document.getElementById(nextId)?.focus();
  }
}

async function handleCarNumber(contentId) {
  const a = document.getElementById(`car-a-${contentId}`)?.value.trim();
  const b = document.getElementById(`car-b-${contentId}`)?.value.trim();
  const c = document.getElementById(`car-c-${contentId}`)?.value.trim();

  if (!a || !b || !c) { alert("차량번호를 모두 입력하세요."); return; }
  if (!/^\d{4}$/.test(c)) { alert("마지막 숫자는 4자리여야 합니다."); return; }
  if (b.length !== 1) { alert("한글 1자를 입력하세요."); return; }

  const text = `${a} ${b} ${c}`;
  await _submitConfirm(contentId, "car_number", text);
}

// ── [T3] 실손 안 선택 ──
const _selectedPlan = {};  // contentId → { label, title }

function selectPlan(contentId, cardEl) {
  // 카드 선택 강조
  document.querySelectorAll(`#plan-${contentId} .plan-card`).forEach(c => {
    c.classList.remove("plan-card--selected");
  });
  cardEl.classList.add("plan-card--selected");
  _selectedPlan[contentId] = {
    label: cardEl.dataset.label,
    title: cardEl.dataset.title,
  };
  // 제출 버튼 활성화
  const btn = document.getElementById(`btn-plan-${contentId}`);
  if (btn) { btn.style.opacity = "1"; btn.style.pointerEvents = "auto"; }
}

async function handlePlanSubmit(contentId) {
  const plan = _selectedPlan[contentId];
  if (!plan) { alert("안을 선택하세요."); return; }
  const text = `${plan.label} - ${plan.title}`;
  await _submitConfirm(contentId, "plan_select", text);
}

// ── [T4] 외부 링크 열람 확인 (파일 다운로드는 서버 측에서 자동 처리) ──
async function markEduViewed(contentId) {
  await _submitConfirm(contentId, "education", "열람 완료");
}

// ── 공통 confirm 제출 ──
async function _submitConfirm(contentId, answerType, text) {
  const res = await fetch(`/api/content/${contentId}/confirm`, {
    method: "PATCH",
    credentials: "include",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ confirmed: true, answer_type: answerType, text }),
  });
  if (res.ok) {
    // 카드 내 응답 영역을 완료 상태로 즉시 교체
    const card = document.querySelector(`[data-content-id="${contentId}"]`);
    const responseArea = card?.querySelector(
      ".yesno-wrap, .choice-wrap, .car-number-wrap, .plan-grid, .edu-card, textarea, [data-plan-submit]"
    );
    const submitBtn = card?.querySelector(".btn-primary");
    if (responseArea) responseArea.outerHTML =
      `<div class="confirmed-text-box">응답: <strong>${text}</strong>
       <span class="badge-confirmed">✅ 완료</span></div>`;
    submitBtn?.remove();
  } else {
    alert("저장에 실패했습니다. 다시 시도해주세요.");
  }
}
```

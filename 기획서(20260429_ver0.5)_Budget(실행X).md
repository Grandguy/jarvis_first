# [PATCH] 관리자 업무추진비 집행/잔액 현황 대시보드

> **접근 권한:** `is_admin = TRUE` 사용자 중 `role = manager` 이상  
> **수정 파일:** `master.xlsx(신규 시트 3개)`, `models/budget.py(신규)`, `repositories/budget_repository.py(신규)`, `routers/budget.py(신규)`, `admin_layer.html`, `admin.js`, `main.css`

---

## 1. 설계 원칙

```
┌─ 접근 제어 ───────────────────────────────────────────────────┐
│  is_admin=TRUE + role=manager|director|executive 만 열람 가능  │
│  일반 관리자(is_admin=TRUE, role=staff)는 해당 메뉴 비표시     │
└────────────────────────────────────────────────────────────────┘

┌─ 할당액 산정 기준 ─────────────────────────────────────────────┐
│  연간 할당액 = 직책별 월 기준액 × 재임 개월 수                 │
│  발령일이 월 중간인 경우 해당 월은 일할 계산                   │
│  예) 발령일 3월 15일, 월 기준액 500,000원                      │
│      → 3월 할당액 = 500,000 × (16/31) = 258,065원             │
└────────────────────────────────────────────────────────────────┘

┌─ 데이터 흐름 ─────────────────────────────────────────────────┐
│  [엑셀 업로드] → [정규화(Normalizer)] → [budget_executions 저장]│
│  → [월별 집행액 집계] → [대시보드 렌더링]                      │
└────────────────────────────────────────────────────────────────┘
```

```python
# app/models/budget.py
from pydantic import BaseModel
from typing import Literal

class ManagerProfile(BaseModel):
    emp_id:           str
    role:             Literal["manager", "director", "executive"]
    appointment_date: str          # "YYYY-MM-DD"
    monthly_budget:   int
    department:       str

class BudgetExecution(BaseModel):
    execution_id:   str
    emp_id:         str
    execution_date: str
    year:           int
    month:          int
    amount:         int
    description:    str
    category:       str
    receipt_no:     str = ""

class MonthlyBudgetSummary(BaseModel):
    year:          int
    month:         int
    allocated:     int    # 해당 월 할당액 (발령일 기준 일할 계산 포함)
    executed:      int    # 해당 월 집행액 합계
    balance:       int    # 잔액 = allocated - executed
    rate:          float  # 집행률 % = executed / allocated × 100

class AnnualBudgetSummary(BaseModel):
    year:               int
    total_allocated:    int
    total_executed:     int
    total_balance:      int
    execution_rate:     float
    monthly_summaries:  list[MonthlyBudgetSummary]
```

## 4. `app/repositories/budget_repository.py` — 신규 파일 생성

```python
# app/repositories/budget_repository.py
import uuid, json, threading
import pandas as pd
from datetime import date, datetime
from calendar import monthrange
from app.core.config import settings

_lock = threading.Lock()

def _load(sheet: str) -> pd.DataFrame:
    return pd.read_excel(settings.EXCEL_PATH, sheet_name=sheet, dtype=str)

def _save(sheet: str, df: pd.DataFrame):
    import openpyxl
    with _lock:
        wb = openpyxl.load_workbook(settings.EXCEL_PATH)
        if sheet in wb.sheetnames:
            del wb[sheet]
        ws = wb.create_sheet(sheet)
        for ci, col in enumerate(df.columns, 1):
            ws.cell(row=1, column=ci, value=col)
        for ri, row in enumerate(df.itertuples(index=False), 2):
            for ci, val in enumerate(row, 1):
                ws.cell(row=ri, column=ci, value=val)
        wb.save(settings.EXCEL_PATH)


# ── 관리자 프로필 ──────────────────────────────────────────────
def get_manager_profile(emp_id: str) -> dict | None:
    df = _load("manager_profiles")
    row = df[df["emp_id"] == emp_id]
    if row.empty:
        return None
    r = row.iloc[0].to_dict()
    r["monthly_budget"] = int(r["monthly_budget"])
    return r


# ── 월별 할당액 계산 (발령일 기준 일할 포함) ──────────────────
def calc_monthly_allocated(profile: dict, year: int, month: int) -> int:
    appt = date.fromisoformat(profile["appointment_date"])
    monthly = profile["monthly_budget"]

    # 발령 전 월이면 0
    if (year, month) < (appt.year, appt.month):
        return 0
    # 발령월: 일할 계산
    if year == appt.year and month == appt.month:
        days_in_month = monthrange(year, month)[1]
        remaining_days = days_in_month - appt.day + 1
        return round(monthly * remaining_days / days_in_month)
    return monthly


# ── 연간 요약 조회 ─────────────────────────────────────────────
def get_annual_summary(emp_id: str, year: int) -> dict:
    profile = get_manager_profile(emp_id)
    if not profile:
        return {}

    df_exec = _load("budget_executions")
    df_year = df_exec[
        (df_exec["emp_id"] == emp_id) &
        (df_exec["year"] == str(year))
    ].copy()
    df_year["amount"] = pd.to_numeric(df_year["amount"], errors="coerce").fillna(0).astype(int)

    monthly_summaries = []
    total_allocated = 0
    total_executed  = 0

    for m in range(1, 13):
        allocated = calc_monthly_allocated(profile, year, m)
        executed  = int(df_year[df_year["month"] == str(m)]["amount"].sum())
        balance   = allocated - executed
        rate      = round(executed / allocated * 100, 1) if allocated > 0 else 0.0

        total_allocated += allocated
        total_executed  += executed

        monthly_summaries.append({
            "year": year, "month": m,
            "allocated": allocated, "executed": executed,
            "balance": balance, "rate": rate,
        })

    return {
        "year":             year,
        "total_allocated":  total_allocated,
        "total_executed":   total_executed,
        "total_balance":    total_allocated - total_executed,
        "execution_rate":   round(total_executed / total_allocated * 100, 1)
                            if total_allocated > 0 else 0.0,
        "monthly_summaries": monthly_summaries,
        "profile":           profile,
    }


# ── 집행 내역 상세 조회 ────────────────────────────────────────
def get_executions(emp_id: str, year: int, month: int | None = None) -> list[dict]:
    df = _load("budget_executions")
    mask = (df["emp_id"] == emp_id) & (df["year"] == str(year))
    if month:
        mask &= (df["month"] == str(month))
    return df[mask].to_dict(orient="records")


# ── 엑셀 업로드 정규화 및 적재 ────────────────────────────────
def normalize_and_save(emp_id: str, raw_df: pd.DataFrame, filename: str) -> dict:
    """
    업로드된 엑셀을 정규화하여 budget_executions 시트에 적재.
    
    기대 컬럼 (유연 매핑 지원):
      날짜/집행일/일자  →  execution_date
      금액/집행금액/원  →  amount
      내용/집행내용/적요 → description
      분류/구분         →  category
      영수증/증빙번호   →  receipt_no (선택)
    """
    COL_MAP = {
        "날짜":   "execution_date", "집행일": "execution_date", "일자": "execution_date",
        "금액":   "amount",          "집행금액": "amount",       "원":   "amount",
        "내용":   "description",     "집행내용": "description",  "적요": "description",
        "분류":   "category",        "구분":    "category",
        "영수증": "receipt_no",      "증빙번호": "receipt_no",
    }
    REQUIRED = {"execution_date", "amount", "description"}

    # 컬럼명 정규화
    raw_df.columns = [str(c).strip() for c in raw_df.columns]
    mapped = {COL_MAP.get(c, c): c for c in raw_df.columns}
    raw_df = raw_df.rename(columns={v: k for k, v in mapped.items() if k in COL_MAP.values()})

    missing = REQUIRED - set(raw_df.columns)
    if missing:
        return {"status": "failed", "error": f"필수 컬럼 없음: {missing}", "row_count": 0}

    errors = []
    rows   = []

    for idx, row in raw_df.iterrows():
        try:
            exec_date = pd.to_datetime(str(row["execution_date"])).date()
            amount    = int(str(row["amount"]).replace(",", "").replace("원", "").strip())
            desc      = str(row.get("description", "")).strip()
            category  = str(row.get("category", "기타")).strip() or "기타"
            receipt   = str(row.get("receipt_no", "")).strip()

            rows.append({
                "execution_id":   str(uuid.uuid4()),
                "emp_id":         emp_id,
                "execution_date": exec_date.isoformat(),
                "year":           exec_date.year,
                "month":          exec_date.month,
                "amount":         amount,
                "description":    desc,
                "category":       category,
                "receipt_no":     receipt,
                "uploaded_at":    datetime.now().isoformat(),
                "source_file":    filename,
            })
        except Exception as e:
            errors.append(f"행 {idx + 2}: {e}")

    if rows:
        df_exist = _load("budget_executions")
        # 동일 파일 재업로드 시 기존 항목 제거 (중복 방지)
        df_exist = df_exist[df_exist["source_file"] != filename]
        df_new   = pd.DataFrame(rows)
        df_merged = pd.concat([df_exist, df_new], ignore_index=True)
        _save("budget_executions", df_merged)

    # 업로드 이력 기록
    upload_row = {
        "upload_id":   str(uuid.uuid4()),
        "emp_id":      emp_id,
        "filename":    filename,
        "uploaded_at": datetime.now().isoformat(),
        "row_count":   len(rows),
        "status":      "success" if not errors else ("partial" if rows else "failed"),
        "error_log":   " | ".join(errors[:10]),  # 최대 10건만 기록
    }
    df_ul = _load("budget_uploads")
    df_ul = pd.concat([df_ul, pd.DataFrame([upload_row])], ignore_index=True)
    _save("budget_uploads", df_ul)

    return {
        "status":    upload_row["status"],
        "row_count": len(rows),
        "errors":    errors,
    }
```

## 5. `app/routers/budget.py` — 신규 파일 생성

```python
# app/routers/budget.py
import io
import pandas as pd
from fastapi import APIRouter, Depends, HTTPException, UploadFile, File, Query
from app.core.dependencies import require_authenticated_user
from app.repositories import budget_repository as repo

router = APIRouter()

def _require_manager(user=Depends(require_authenticated_user)):
    """manager_profiles에 등록된 관리자만 허용"""
    profile = repo.get_manager_profile(user["sub"])
    if not profile:
        raise HTTPException(status_code=403, detail="업무추진비 조회 권한이 없습니다.")
    return user

# 연간 요약 조회
@router.get("/api/budget/summary")
async def get_budget_summary(
    year: int = Query(..., description="조회 연도"),
    user=Depends(_require_manager)
):
    return repo.get_annual_summary(user["sub"], year)

# 월별 집행 내역 상세 조회
@router.get("/api/budget/executions")
async def get_executions(
    year:  int           = Query(...),
    month: int | None    = Query(None),
    user=Depends(_require_manager)
):
    return repo.get_executions(user["sub"], year, month)

# 엑셀 업로드 및 정규화
@router.post("/api/budget/upload")
async def upload_budget_excel(
    file: UploadFile = File(...),
    user=Depends(_require_manager)
):
    if not file.filename.endswith((".xlsx", ".xls")):
        raise HTTPException(status_code=400, detail="xlsx 또는 xls 파일만 허용됩니다.")
    contents = await file.read()
    try:
        raw_df = pd.read_excel(io.BytesIO(contents))
    except Exception as e:
        raise HTTPException(status_code=422, detail=f"파일 파싱 오류: {e}")

    result = repo.normalize_and_save(user["sub"], raw_df, file.filename)
    return result

# 업로드 이력
@router.get("/api/budget/uploads")
async def get_upload_history(user=Depends(_require_manager)):
    from app.repositories.budget_repository import _load
    df = _load("budget_uploads")
    return df[df["emp_id"] == user["sub"]].to_dict(orient="records")
```

```javascript
let _budgetData = null;   // 연간 요약 캐시

// ── 초기화: 연도 선택자 세팅 ──
function initBudgetPanel() {
  const sel = document.getElementById("budget-year-select");
  const now = new Date().getFullYear();
  sel.innerHTML = "";
  for (let y = now; y >= now - 2; y--) {
    const opt = document.createElement("option");
    opt.value = y; opt.textContent = `${y}년`;
    sel.appendChild(opt);
  }
  loadBudgetDashboard();
  loadUploadHistory();
}

// ── 연간 데이터 로드 및 렌더 ──
async function loadBudgetDashboard() {
  const year = document.getElementById("budget-year-select").value;
  const res  = await fetch(`/api/budget/summary?year=${year}`, { credentials: "include" });
  if (!res.ok) { alert("조회 권한이 없거나 데이터가 없습니다."); return; }

  _budgetData = await res.json();
  renderSummaryCards(_budgetData);
  renderBarChart(_budgetData.monthly_summaries);
  renderMonthlyTable(_budgetData.monthly_summaries);

  // 프로필 라벨
  const p = _budgetData.profile;
  document.getElementById("budget-profile-label").textContent =
    `${p.department} | 발령일: ${p.appointment_date} | 월 기준액: ${fmt(p.monthly_budget)}원`;
}

// ── 요약 카드 ──
function renderSummaryCards(data) {
  const bal = data.total_balance;
  document.getElementById("bc-allocated").textContent = fmt(data.total_allocated) + "원";
  document.getElementById("bc-executed").textContent  = fmt(data.total_executed)  + "원";
  document.getElementById("bc-balance").textContent   = fmt(Math.abs(bal))        + "원" + (bal < 0 ? " 초과" : "");
  document.getElementById("bc-balance").style.color   = bal < 0 ? "#f03e3e" : "inherit";
  document.getElementById("bc-rate").textContent      = data.execution_rate + "%";
  document.getElementById("bc-balance-pct").textContent =
    data.total_allocated > 0 ? `할당액 대비 ${(100 - data.execution_rate).toFixed(1)}% 남음` : "";

  const bar = document.getElementById("bc-progress-bar");
  const pct = Math.min(data.execution_rate, 100);
  bar.style.width = pct + "%";
  bar.style.background = pct >= 90 ? "#f03e3e" : pct >= 70 ? "#f59f00" : "#3b5bdb";
}

// ── 바 차트 (CSS 기반, 외부 라이브러리 불필요) ──
function renderBarChart(months) {
  const maxVal = Math.max(...months.map(m => Math.max(m.allocated, m.executed)), 1);
  const MONTH_KO = ["1월","2월","3월","4월","5월","6월","7월","8월","9월","10월","11월","12월"];

  document.getElementById("budget-bar-chart").innerHTML = months.map((m, i) => `
    <div class="chart-col" onclick="openDrawer(${m.year}, ${m.month})">
      <div class="chart-bars">
        <div class="chart-bar chart-bar--alloc"
             style="height:${Math.round(m.allocated / maxVal * 140)}px"
             title="할당액 ${fmt(m.allocated)}원"></div>
        <div class="chart-bar chart-bar--exec ${m.executed > m.allocated ? 'chart-bar--over' : ''}"
             style="height:${Math.round(m.executed / maxVal * 140)}px"
             title="집행액 ${fmt(m.executed)}원"></div>
      </div>
      <span class="chart-label">${MONTH_KO[i]}</span>
      <span class="chart-rate">${m.allocated > 0 ? m.rate + "%" : "-"}</span>
    </div>
  `).join("");
}

// ── 월별 테이블 ──
function renderMonthlyTable(months) {
  const MONTH_KO = ["1월","2월","3월","4월","5월","6월","7월","8월","9월","10월","11월","12월"];
  const now = new Date();

  document.getElementById("budget-monthly-tbody").innerHTML = months.map(m => {
    const isPast    = m.year < now.getFullYear() || (m.year === now.getFullYear() && m.month < now.getMonth() + 1);
    const isCurrent = m.year === now.getFullYear() && m.month === now.getMonth() + 1;
    const hasAlloc  = m.allocated > 0;
    const isOver    = m.executed > m.allocated && hasAlloc;

    const statusBadge = !hasAlloc     ? `<span class="badge badge--gray">발령 전</span>`
                      : isOver        ? `<span class="badge badge--red">초과</span>`
                      : isCurrent     ? `<span class="badge badge--blue">진행 중</span>`
                      : m.rate >= 100 ? `<span class="badge badge--orange">완전 소진</span>`
                      : isPast        ? `<span class="badge badge--green">완료</span>`
                      :                 `<span class="badge badge--gray">예정</span>`;

    return `<tr class="${isCurrent ? 'row--current' : ''}">
      <td style="font-weight:${isCurrent ? 700 : 400}">${MONTH_KO[m.month - 1]}</td>
      <td style="text-align:right;">${hasAlloc ? fmt(m.allocated) + "원" : "-"}</td>
      <td style="text-align:right; color:${m.executed > 0 ? '#f03e3e' : '#999'}">
        ${m.executed > 0 ? fmt(m.executed) + "원" : "-"}
      </td>
      <td style="text-align:right; color:${isOver ? '#f03e3e' : '#2e7d32'}; font-weight:600;">
        ${hasAlloc ? (isOver ? "-" : "") + fmt(Math.abs(m.balance)) + "원" : "-"}
      </td>
      <td style="text-align:right;">${hasAlloc ? m.rate + "%" : "-"}</td>
      <td style="text-align:center;">${statusBadge}</td>
      <td style="text-align:center;">
        ${hasAlloc ? `<button class="btn-link" onclick="openDrawer(${m.year},${m.month})">내역 보기</button>` : "-"}
      </td>
    </tr>`;
  }).join("");
}

// ── 월별 상세 드로어 ──
async function openDrawer(year, month) {
  const MONTH_KO = ["1월","2월","3월","4월","5월","6월","7월","8월","9월","10월","11월","12월"];
  document.getElementById("drawer-title").textContent = `${year}년 ${MONTH_KO[month-1]} 집행 내역`;

  const res  = await fetch(`/api/budget/executions?year=${year}&month=${month}`, { credentials: "include" });
  const rows = await res.json();

  document.getElementById("drawer-tbody").innerHTML = rows.length === 0
    ? `<tr><td colspan="5" style="text-align:center;color:#999;padding:20px;">집행 내역이 없습니다.</td></tr>`
    : rows.map(r => `
        <tr>
          <td>${r.execution_date}</td>
          <td style="text-align:right; font-weight:600;">${fmt(parseInt(r.amount))}원</td>
          <td><span class="badge badge--blue">${r.category}</span></td>
          <td>${r.description}</td>
          <td style="color:#999;">${r.receipt_no || "-"}</td>
        </tr>`).join("");

  document.getElementById("budget-detail-drawer").classList.remove("hidden");
}

function closeDrawer() {
  document.getElementById("budget-detail-drawer").classList.add("hidden");
}

// ── 엑셀 업로드 ──
async function uploadBudgetExcel(input) {
  const file = input.files[0];
  if (!file) return;

  const fd = new FormData();
  fd.append("file", file);
  const res  = await fetch("/api/budget/upload", { method: "POST", credentials: "include", body: fd });
  const data = await res.json();

  if (data.status === "failed") {
    alert(`업로드 실패: ${data.error}`);
  } else {
    const msg = data.status === "partial"
      ? `${data.row_count}건 적재 완료 (일부 오류: ${data.errors?.length}건)`
      : `${data.row_count}건 정상 적재 완료`;
    alert(msg);
    await loadBudgetDashboard();
    await loadUploadHistory();
  }
  input.value = "";  // 파일 입력 초기화
}

// ── 업로드 이력 ──
async function loadUploadHistory() {
  const res  = await fetch("/api/budget/uploads", { credentials: "include" });
  const rows = await res.json();
  const STATUS = { success: ["완료","badge--green"], partial: ["부분 완료","badge--orange"], failed: ["실패","badge--red"] };

  document.getElementById("upload-history-tbody").innerHTML = rows.length === 0
    ? `<tr><td colspan="5" style="text-align:center;color:#999;">업로드 이력이 없습니다.</td></tr>`
    : rows.slice().reverse().map(r => {
        const [label, cls] = STATUS[r.status] ?? ["알 수 없음", "badge--gray"];
        return `<tr>
          <td>${r.filename}</td>
          <td>${r.uploaded_at?.slice(0,16).replace("T"," ") ?? "-"}</td>
          <td style="text-align:center;">${r.row_count}건</td>
          <td style="text-align:center;"><span class="badge ${cls}">${label}</span></td>
          <td style="font-size:12px; color:#888;">${r.error_log || "-"}</td>
        </tr>`;
      }).join("");
}

// ── CSV 내보내기 ──
function exportBudgetCSV() {
  if (!_budgetData) return;
  const year = _budgetData.year;
  const MONTH_KO = ["1월","2월","3월","4월","5월","6월","7월","8월","9월","10월","11월","12월"];
  const header = "월,할당액,집행액,잔액,집행률\n";
  const body   = _budgetData.monthly_summaries.map(m =>
    `${MONTH_KO[m.month-1]},${m.allocated},${m.executed},${m.balance},${m.rate}%`
  ).join("\n");

  const blob = new Blob(["\uFEFF" + header + body], { type: "text/csv;charset=utf-8;" });
  const a = document.createElement("a");
  a.href = URL.createObjectURL(blob);
  a.download = `업무추진비_${year}.csv`;
  a.click();
}

// ── 숫자 포맷 ──
function fmt(n) {
  return Number(n).toLocaleString("ko-KR");
}
```

```python
# app/repositories/base_repository.py
from abc import ABC, abstractmethod
from typing import Optional
from app.models.user import UserInDB
from app.models.content import Content, ContentCreate

class BaseUserRepository(ABC):
    @abstractmethod
    def get_user_by_emp_id(self, emp_id: str) -> Optional[UserInDB]: ...

    @abstractmethod
    def list_all_users(self) -> list[UserInDB]: ...

class BaseContentRepository(ABC):
    @abstractmethod
    def get_contents_by_emp_id(self, emp_id: str) -> list[Content]: ...

    @abstractmethod
    def create_content(self, payload: ContentCreate, creator_id: str) -> Content: ...

    @abstractmethod
    def delete_content(self, content_id: str) -> bool: ...

    @abstractmethod
    def set_confirmation(self, content_id: str, emp_id: str, confirmed: bool) -> bool: ...
```

```python
# app/repositories/excel_repository.py
import uuid, pandas as pd
from datetime import datetime
from app.repositories.base_repository import BaseUserRepository, BaseContentRepository
from app.core.config import settings

class ExcelUserRepository(BaseUserRepository):
    def __init__(self):
        self._path = settings.EXCEL_PATH

    def _load(self, sheet: str) -> pd.DataFrame:
        return pd.read_excel(self._path, sheet_name=sheet, dtype=str)

    def _save(self, sheet: str, df: pd.DataFrame):
        # openpyxl 엔진으로 특정 시트만 덮어쓰기
        import openpyxl
        wb = openpyxl.load_workbook(self._path)
        if sheet in wb.sheetnames:
            del wb[sheet]
        ws = wb.create_sheet(sheet)
        for col_idx, col_name in enumerate(df.columns, 1):
            ws.cell(row=1, column=col_idx, value=col_name)
        for row_idx, row in enumerate(df.itertuples(index=False), 2):
            for col_idx, val in enumerate(row, 1):
                ws.cell(row=row_idx, column=col_idx, value=val)
        wb.save(self._path)

    def get_user_by_emp_id(self, emp_id: str):
        df = self._load("users")
        row = df[df["emp_id"] == emp_id]
        if row.empty:
            return None
        r = row.iloc[0]
        return UserInDB(
            emp_id=r["emp_id"], name=r["name"],
            email=r["email"],
            is_admin=(str(r["is_admin"]).upper() == "TRUE"),
            is_active=(str(r["is_active"]).upper() == "TRUE"),
        )

    def list_all_users(self):
        df = self._load("users")
        return [self.get_user_by_emp_id(eid) for eid in df["emp_id"]]
```


```
        ▼

│  { "emp_id": "EMP001", "otp": "4821" }│
│                                       │
│  1. 캐시에서 emp_id의 OTP 조회        │
```

### 3-2. OTP 서비스 핵심 코드

```python
# app/services/otp_service.py
import secrets, smtplib, time
from email.mime.text import MIMEText
from app.core.config import settings

# 메모리 캐시 (Phase 1; Phase 2에서 Redis로 교체)
_otp_store: dict[str, dict] = {}
_attempt_store: dict[str, int] = {}

MAX_ATTEMPTS = 3
OTP_TTL = 300  # 5분

def generate_and_store_otp(emp_id: str) -> str:
    otp = f"{secrets.randbelow(1_000_000):06d}"
    _otp_store[emp_id] = {"otp": otp, "created_at": time.time()}
    _attempt_store[emp_id] = 0
    return otp

def send_otp_email(email: str, otp: str, name: str):
    body = f"""안녕하세요, {name}님.\n\n인증 코드: {otp}\n\n본 코드는 5분간 유효합니다."""
    msg = MIMEText(body, "plain", "utf-8")
    msg["Subject"] = "[시스템] 이메일 인증 코드 안내"
    msg["From"] = settings.SMTP_FROM
    msg["To"] = email
    with smtplib.SMTP(settings.SMTP_HOST, settings.SMTP_PORT) as s:
        s.starttls()
        s.login(settings.SMTP_USER, settings.SMTP_PASSWORD)
        s.sendmail(settings.SMTP_FROM, [email], msg.as_string())

def verify_otp(emp_id: str, input_otp: str) -> tuple[bool, str]:
    """반환: (성공 여부, 실패 사유)"""
    record = _otp_store.get(emp_id)
    if not record:
        return False, "expired"
    if time.time() - record["created_at"] > OTP_TTL:
        del _otp_store[emp_id]
        return False, "expired"
    attempts = _attempt_store.get(emp_id, 0)
    if attempts >= MAX_ATTEMPTS:
        return False, "locked"
    if record["otp"] != input_otp:
        _attempt_store[emp_id] = attempts + 1
        return False, "mismatch"
    del _otp_store[emp_id]
    del _attempt_store[emp_id]
    return True, "ok"
```

### 3-3. 세션 토큰 (JWT) 발급

```python
# app/services/session_service.py
import jwt
from datetime import datetime, timedelta, timezone
from app.core.config import settings

ALGORITHM = "HS256"
TOKEN_EXPIRE_HOURS = 8

def create_session_token(emp_id: str, is_admin: bool) -> str:
    payload = {
        "sub": emp_id,
        "is_admin": is_admin,
        "exp": datetime.now(timezone.utc) + timedelta(hours=TOKEN_EXPIRE_HOURS),
        "iat": datetime.now(timezone.utc),
    }
    return jwt.encode(payload, settings.SECRET_KEY, algorithm=ALGORITHM)

def decode_session_token(token: str) -> dict:
    """만료/변조 시 jwt.ExpiredSignatureError / jwt.InvalidTokenError 발생"""
    return jwt.decode(token, settings.SECRET_KEY, algorithms=[ALGORITHM])
```

## 4. 레이어드 HTML 컴포넌트 구조

### 4-1. 설계 원칙

'Layered HTML' 방식은 단일 `shell.html`이 SPA 컨테이너 역할을 하며, 각 기능 레이어(인증, 컨텐츠, 관리자)를 동적으로 DOM에 주입(Merge)하는 방식이다. 별도의 .html 파일 URL은 사용자에게 노출되지 않는다.

```
shell.html
├── <div id="layer-auth">      ← 미인증 시 표시 (auth_layer fragment)
├── <div id="layer-content">   ← 인증 후 메인 (content_layer fragment)
└── <div id="layer-admin">     ← is_admin=true 시 추가 활성화 (admin_layer fragment)
```

### 4-2. Shell HTML

```html
<!-- app/templates/shell.html -->
<!DOCTYPE html>
<html lang="ko">
<head>
  <meta charset="UTF-8">
  <title>컨텐츠 열람 시스템</title>
  <link rel="stylesheet" href="/static/css/main.css">
</head>
<body>
  <!-- 레이어 마운트 포인트 -->
  <div id="app-root">
    <div id="layer-auth"   class="layer hidden"></div>
    <div id="layer-content" class="layer hidden"></div>
    <div id="layer-admin"  class="layer hidden"></div>
  </div>

  <!-- 전역 상태: 서버 사이드에서 주입 (Jinja2) -->
  <script>
    window.__APP_STATE__ = {
      isAuthenticated: {{ is_authenticated | tojson }},
      isAdmin: {{ is_admin | tojson }},
      empId: {{ emp_id | tojson }}
    };
  </script>

  <script src="/static/js/auth.js"    defer></script>
  <script src="/static/js/content.js" defer></script>
  <script src="/static/js/admin.js"   defer></script>
</body>
</html>
```

### 4-3. 레이어 주입 방식 (Fragment API)

서버는 각 레이어의 HTML 조각(fragment)을 별도 엔드포인트로 제공하되, 직접 접근 시 401을 반환한다.

```python
# app/routers/content.py (Fragment 엔드포인트 예시)
from fastapi import APIRouter, Request, Depends
from fastapi.responses import HTMLResponse
from fastapi.templating import Jinja2Templates
from app.core.dependencies import require_authenticated_user

router = APIRouter()
templates = Jinja2Templates(directory="app/templates")

@router.get("/fragment/content", response_class=HTMLResponse)
async def get_content_fragment(
    request: Request,
    current_user=Depends(require_authenticated_user)
):
    contents = content_service.get_contents_for_user(current_user.emp_id)
    return templates.TemplateResponse(
        "fragments/content_layer.html",
        {"request": request, "contents": contents, "user": current_user}
    )
```

```javascript
// static/js/content.js — 레이어 동적 주입
async function mountLayer(layerId, fragmentUrl) {
  const res = await fetch(fragmentUrl, { credentials: "include" });
  if (!res.ok) return;
  const html = await res.text();
  const target = document.getElementById(layerId);
  target.innerHTML = html;
  target.classList.remove("hidden");
}

document.addEventListener("DOMContentLoaded", async () => {
  const state = window.__APP_STATE__;
  if (!state.isAuthenticated) {
    await mountLayer("layer-auth", "/fragment/auth");
    return;
  }
  await mountLayer("layer-content", "/fragment/content");
  if (state.isAdmin) {
    await mountLayer("layer-admin", "/fragment/admin");
  }
});
```

```javascript
      method: "PATCH",
      credentials: "include",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ confirmed })
    });
    if (!res.ok) {
      checkbox.checked = !confirmed;  // 실패 시 롤백
      alert("저장에 실패했습니다. 다시 시도해주세요.");
    }
  } catch (e) {
    checkbox.checked = !confirmed;
    console.error(e);
  }
}
```

## 5. 보안 전략

### 5-1. 경로 은닉 (Path Obfuscation)

| 위협 | 대응 전략 |
|---|---|
| 직접 URL 추측 (`.html` 노출) | 모든 라우팅을 FastAPI 동적 라우터로 처리, 정적 파일 서빙 비활성화 |
| Fragment URL 직접 접근 | 모든 fragment 엔드포인트에 `require_authenticated_user` 의존성 적용 |
| 데이터 파일 직접 접근 | `data/` 디렉터리를 웹 루트 밖에 위치. `EXCEL_PATH`는 환경 변수로만 관리 |
| 경로 패턴 노출 | 외부 공개 URL은 의미없는 동사형 경로만 사용 (`/app`, `/auth/step1`) |

**URL 설계 원칙:**

```
# 노출되는 URL (외부)        # 실제 내부 처리
GET  /                    →  index redirect to /app
GET  /app                 →  shell.html 렌더 (인증 여부 판단)
POST /auth/step1          →  OTP 요청 (내부: request_otp)
POST /auth/step2          →  OTP 검증 (내부: verify_otp)
GET  /fragment/content    →  content_layer.html (세션 필수)
GET  /fragment/admin      →  admin_layer.html (관리자 세션 필수)
```

```python
from pydantic import BaseModel, field_validator
import re

class OTPRequest(BaseModel):
    emp_id: str
    otp: str

    @field_validator("emp_id")
    @classmethod

        return v
```

```python
from sqlalchemy import Column, String, Boolean, DateTime, ForeignKey, Text
from sqlalchemy.ext.declarative import declarative_base
from datetime import datetime, timezone

Base = declarative_base()

class User(Base):
    __tablename__ = "users"
    emp_id     = Column(String(20), primary_key=True)
    name       = Column(String(100), nullable=False)
    email      = Column(String(255), nullable=False, unique=True)
    is_admin   = Column(Boolean, default=False, nullable=False)
    is_active  = Column(Boolean, default=True, nullable=False)
    created_at = Column(DateTime(timezone=True), default=lambda: datetime.now(timezone.utc))

class Content(Base):
    __tablename__ = "contents"
    content_id     = Column(String(36), primary_key=True)  # UUID v4
    title          = Column(String(500), nullable=False)
    target_emp_id  = Column(String(20), ForeignKey("users.emp_id"), nullable=False)
    body_1         = Column(Text, nullable=False)
    body_2         = Column(Text, nullable=True)
    body_3         = Column(Text, nullable=True)
    created_by     = Column(String(20), ForeignKey("users.emp_id"), nullable=False)
    created_at     = Column(DateTime(timezone=True), default=lambda: datetime.now(timezone.utc))
    updated_at     = Column(DateTime(timezone=True), nullable=True)

class Confirmation(Base):
    __tablename__ = "confirmations"
    confirmation_id = Column(String(36), primary_key=True)
    content_id      = Column(String(36), ForeignKey("contents.content_id"), nullable=False)
    emp_id          = Column(String(20), ForeignKey("users.emp_id"), nullable=False)
    is_confirmed    = Column(Boolean, default=False, nullable=False)
    confirmed_at    = Column(DateTime(timezone=True), nullable=True)
```
